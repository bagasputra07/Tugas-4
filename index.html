🧬 1. Inheritance (Pewarisan)

extends keyword digunakan untuk mewarisi class parent. super() memanggil constructor parent, dan @override untuk mengubah implementasi method.

📌 Contoh Dasar: Hewan → Kucing

// Nama: Azriel Miracle Yuandre
// Nim: 251410024
// Kelas: SI2A

class Hewan {
  void bersuara() => print('...');
}

class Kucing extends Hewan {
  @override
  void bersuara() => print('Meong');
}

void main() {
  Hewan hewan = Hewan();
  Kucing kucing = Kucing();

  hewan.bersuara(); // Output: ...
  kucing.bersuara(); // Output: Meong
}
💡 Catatan: Dengan inheritance, subclass mewarisi semua method dan properti dari superclass, lalu bisa menambah atau mengubah perilaku.

🎭 2. Polimorfisme (Banyak Bentuk)

Objek dari subclass dapat diperlakukan sebagai objek superclass. Dart mendukung runtime polymorphism melalui @override.

🔹 Bentuk → Lingkaran & Persegi

class Bentuk {
  void gambar() => print('Menggambar bentuk');
}
class Lingkaran extends Bentuk {
  @override void gambar() => print('Menggambar lingkaran');
}
class Persegi extends Bentuk {
  @override void gambar() => print('Menggambar persegi');
}
void main() {
  Bentuk bentuk1 = Lingkaran();
  Bentuk bentuk2 = Persegi();
  bentuk1.gambar(); // lingkaran
  bentuk2.gambar(); // persegi
}

🔹 Manfaat Polimorfisme

  • ✔ Fleksibilitas: kode bekerja dengan berbagai tipe
  • ✔ Extensibility: mudah menambah tipe baru
  • ✔ Code Reusability: method umum di superclass
  • ✔ Maintainability: perubahan lebih mudah dikelola

🔍 Operator is dan as

class Kendaraan {}
class Mobil extends Kendaraan {
  void klakson() => print('Tin!');
}
void main() {
  Kendaraan k = Mobil();
  if (k is Mobil) {
    k.klakson();   // aman dengan is
  }
  (k as Mobil).klakson();   // casting langsung
  (k as Mobil?)?.klakson(); // null-aware casting
}

📦 3. Abstraksi & Abstract Class

Menyembunyikan detail implementasi, hanya menampilkan fungsionalitas. Abstract class tidak bisa diinstansiasi langsung, bisa memiliki method abstract (tanpa tubuh) dan concrete.

abstract class Kendaraan {
  String? merk;
  void start();      // method abstract
  void stop() {      // concrete method
    print('Kendaraan berhenti');
  }
}
class Mobil extends Kendaraan {
  @override void start() => print('Mobil ${merk ?? 'tanpa merk'} menyala');
  @override void stop() => print('Mobil ${merk ?? 'tanpa merk'} berhenti');
}
void main() {
  Mobil mobil1 = Mobil()..merk = 'Toyota';
  mobil1.start(); // Mobil Toyota menyala
}

🧩 Abstract Method & Interface di Dart

Method abstract wajib di-override oleh subclass. Sejak Dart 3.0, interface class eksplisit tersedia, namun setiap class bisa menjadi implicit interface dengan keyword implements.

class Gitar extends AlatMusik {

⚙️ 4. Static Members & Method

Properti atau method yang dimiliki oleh class itu sendiri, diakses langsung via ClassName.member tanpa instansiasi.

class MathUtils {
  static double pi = 3.14;
  static int add(int x, int y) => x + y;
}
void main() {
  print(MathUtils.pi);     // 3.14
  print(MathUtils.add(10, 20)); // 30
}

📌 Rangkuman Konsep Kunci

✔ Inheritance

class Anak extends Induk → mewarisi properti & method, bisa @override.

✔ Polimorfisme

Objek subclass disimpan dalam tipe superclass, method yang dipanggil sesuai objek asli.

✔ Abstract Class

Gunakan abstract class untuk mendefinisikan template, tidak bisa di-instansiasi.

✔ Interface (implements)

Class bisa mengimplementasi banyak interface, harus mengimplement ulang semua method.

✔ Static Member

Dimiliki oleh class, diakses langsung tanpa objek: NamaClass.method().

✨ Best Practice: Gunakan extends untuk hubungan IS-A (hierarki natural), gunakan implements untuk kontrak perilaku (CAN-DO). Kombinasi abstraksi dan polimorfisme membuat kode lebih fleksibel dan mudah dikembangkan.

🧮 Soal 5: MathUtils Extended — factorial, prima, median, random 
class MathLovely {
  static int faktorial(int n) {
    if (n < 0) throw Exception('no negatif');
    return n <= 1 ? 1 : n * faktorial(n-1);
  }
  static bool isPrima(int n) {
    if (n < 2) return false;
    for (int i = 2; i * i <= n; i++) if (n % i == 0) return false;
    return true;
  }
  static double rataRata(List<num> angka) {
    if (angka.isEmpty) return 0;
    return angka.reduce((a,b) => a+b) / angka.length;
  }
  static double median(List<num> data) {
    List<num> sorted = List.from(data)..sort();
    int mid = sorted.length ~/ 2;
    if (sorted.length % 2 == 0) return (sorted[mid-1] + sorted[mid]) / 2;
    return sorted[mid];
  }
  static int randomInt(int min, int max) => min + DateTime.now().millisecond % (max - min + 1);
}

void main() {
  print('Faktorial 7: ${MathLovely.faktorial(7)}');
  print('Prima 29? ${MathLovely.isPrima(29)}');
  print('Rata-rata [10,20,30,40]: ${MathLovely.rataRata([10,20,30,40])}');
  print('Median [3,5,1,9,2]: ${MathLovely.median([3,5,1,9,2])}');
}

🎬 Proyek Akhir: MediaPlayer (Audio & Video) + Interface Playable/Pausable 
abstract class MediaStream {
  String judul;
  int durasiDetik;
  MediaStream(this.judul, this.durasiDetik);
  void play();
  void stop();
  void info();
  void displayDuration() {
    int m = durasiDetik ~/ 60;
    int d = durasiDetik % 60;
    print('⏱️ Durasi: ${m.toString().padLeft(2,'0')}:${d.toString().padLeft(2,'0')}');
  }
}

abstract interface class PlayableControl { void play(); void resume(); }
abstract interface class PauseControl { void pause(); void resumeFromPause(); }

class MusicTrack extends MediaStream implements PlayableControl, PauseControl {
  String format;
  int bitrate;
  bool _playing = false, _paused = false;
  MusicTrack(super.judul, super.durasiDetik, this.format, this.bitrate);
  @override void play() { _playing = true; _paused = false; print('🎵 Memutar lagu: $judul ($format)'); }
  @override void pause() { if(_playing){ _playing=false; _paused=true; print('⏸️ Jeda lagu: $judul');} }
  @override void resume() { if(_paused){ _playing=true; _paused=false; print('▶️ Melanjutkan $judul');} }
  @override void resumeFromPause() => resume();
  @override void stop() { _playing=false; _paused=false; print('⏹️ Stop: $judul'); }
  @override void info() { print('🎧 AUDIO: $judul | ${bitrate}kbps | format $format'); displayDuration(); }
}

class MovieClip extends MediaStream implements PlayableControl, PauseControl {
  String resolusi;
  bool _playing = false, _paused = false;
  MovieClip(super.judul, super.durasiDetik, this.resolusi);
  @override void play() { _playing = true; _paused = false; print('🎬 Memutar video: $judul ($resolusi)'); }
  @override void pause() { if(_playing){ _playing=false; _paused=true; print('⏸️ Video dijeda');} }
  @override void resume() { if(_paused){ _playing=true; _paused=false; print('▶️ Resume video $judul');} }
  @override void resumeFromPause() => resume();
  @override void stop() { _playing=false; _paused=false; print('🛑 Video dihentikan: $judul'); }
  @override void info() { print('📽️ VIDEO: $judul | Resolusi $resolusi'); displayDuration(); }
}
