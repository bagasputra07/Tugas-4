Pengantar Polimorfisme

Dart menyediakan beberapa mekanisme inisialisasi objek untuk berbagai kebutuhan arsitektur:

  1. Arti: "Banyak bentuk" (poly = banyak, morph = bentuk)
  2. Konsep: Objek dari subclass dapat diperlakukan sebagai objek dari superclass
  3. Parameterized Constructor: ClassName(this.param)
  4. Contoh: Semua Kucing , Anjing , Burung adalah Hewan
  5. i.Compile-time polymorphism: Method overloading (tidak didukung Dart)
  6. ii. Runtime polymorphism: Method overriding (didukung Dart)

import 'dart:math';

class MathUtils {
  // 1. Menghitung factorial
  static int factorial(int n) {
    if (n < 0) throw ArgumentError('Factorial tidak terdefinisi untuk bilangan negatif');
    if (n == 0 || n == 1) return 1;
    
    int result = 1;
    for (int i = 2; i <= n; i++) {
      result *= i;
    }
    return result;
  }
  
  // 2. Cek bilangan prima
  static bool isPrime(int number) {
    if (number < 2) return false;
    if (number == 2) return true;
    if (number % 2 == 0) return false;
    
    for (int i = 3; i <= sqrt(number); i += 2) {
      if (number % i == 0) return false;
    }
    return true;
  }
  
  // 3. Membulatkan double ke desimal tertentu
  static double roundToDecimal(double value, int decimalPlaces) {
    if (decimalPlaces < 0) throw ArgumentError('decimalPlaces harus >= 0');
    
    double factor = pow(10, decimalPlaces).toDouble();
    return (value * factor).roundToDouble() / factor;
  }
  
  // 4. Menghasilkan bilangan acak integer dalam range [min, max]
  static int randomInt(int min, int max) {
    if (min > max) throw ArgumentError('min harus <= max');
    
    final random = Random();
    return min + random.nextInt(max - min + 1);
  }
  
  // 5. Menghasilkan bilangan acak double dalam range [min, max]
  static double randomDouble(double min, double max) {
    if (min > max) throw ArgumentError('min harus <= max');
    
    final random = Random();
    return min + random.nextDouble() * (max - min);
  }
  
  // 6. Menghitung rata-rata dari list angka
  static double average(List numbers) {
    if (numbers.isEmpty) throw ArgumentError('List tidak boleh kosong');
    
    num sum = 0;
    for (var number in numbers) {
      sum += number;
    }
    return sum / numbers.length;
  }
  
  // 7. Menghitung median dari list angka
  static double median(List numbers) {
    if (numbers.isEmpty) throw ArgumentError('List tidak boleh kosong');
    
    List sortedNumbers = List.from(numbers)..sort();
    int length = sortedNumbers.length;
    
    if (length % 2 == 0) {
      // Jumlah data genap
      int midIndex = length ~/ 2;
      return (sortedNumbers[midIndex - 1] + sortedNumbers[midIndex]) / 2;
    } else {
      // Jumlah data ganjil
      int midIndex = length ~/ 2;
      return sortedNumbers[midIndex].toDouble();
    }
  }
}

void main() {
  // Contoh penggunaan MathUtils
  print('=== MathUtils Demo ===\n');
  
  // 1. Factorial
  print('1. Factorial:');
  print('   factorial(5) = ${MathUtils.factorial(5)}');
  print('   factorial(0) = ${MathUtils.factorial(0)}');
  print('   factorial(7) = ${MathUtils.factorial(7)}\n');
  
  // 2. Bilangan prima
  print('2. Bilangan Prima:');
  print('   isPrime(7) = ${MathUtils.isPrime(7)}');
  print('   isPrime(10) = ${MathUtils.isPrime(10)}');
  print('   isPrime(17) = ${MathUtils.isPrime(17)}\n');
  
  // 3. Pembulatan double
  print('3. Pembulatan Double:');
  print('   roundToDecimal(3.14159, 2) = ${MathUtils.roundToDecimal(3.14159, 2)}');
  print('   roundToDecimal(2.71828, 3) = ${MathUtils.roundToDecimal(2.71828, 3)}\n');
  
  // 4 & 5. Random numbers
  print('4 & 5. Random Numbers:');
  print('   randomInt(1, 10) = ${MathUtils.randomInt(1, 10)}');
  print('   randomDouble(1.0, 5.0) = ${MathUtils.randomDouble(1.0, 5.0).toStringAsFixed(4)}\n');
  
  // 6. Rata-rata
  print('6. Rata-rata:');
  List scores = [85, 90, 78, 92, 88];
  print('   Data: $scores');
  print('   average = ${MathUtils.average(scores).toStringAsFixed(2)}\n');
  
  // 7. Median
  print('7. Median:');
  List data1 = [3, 1, 4, 1, 5, 9, 2];
  List data2 = [10, 20, 30, 40];
  print('   Data ganjil: $data1');
  print('   median = ${MathUtils.median(data1)}');
  print('   Data genap: $data2');
  print('   median = ${MathUtils.median(data2)}\n');
  
  // Demonstrasi error handling
  print('=== Error Handling Demo ===');
  try {
    MathUtils.factorial(-1);
  } catch (e) {
    print('   Error factorial: $e');
  }
  
  try {
    MathUtils.average([]);
  } catch (e) {
    print('   Error average: $e');
  }
}
